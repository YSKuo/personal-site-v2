import { useSiteMetadata } from "@reflexjs/gatsby-theme-core"

export const Copyright = ({ ...props }) => {
  const { title } = useSiteMetadata()
  return (
    <P fontSize="sm" {...props}>
      {
        new Date().getFullYear() === 2020 ? new Date().getFullYear() : `2020 - ${new Date().getFullYear()}`
      }
      {new Date().getFullYear()} {title}
    </P>
  )
}
